34  public static void DecodeUnityClassInfo(Type classType, string package= "UnityEngine", string iconPath= null, string description= null) {
35  string company = "Unity";
36  bool decodeAllPublicMembers= true;
37  if(package == null) package = kUnityEnginePackage;
38  if(description == null) description = "Unity class "+classType.Name;
39  if(iconPath == null) iconPath = kUnityIcon;
40  iCS_Reflection.DecodeClassInfo(classType, company, package, description, iconPath, decodeAllPublicMembers,true);
41  }
42  // ----------------------------------------------------------------------
43  // Use this function to register a Unity message on a class.
44  // Messages are invoked using dynamic (runtime) lookup and are not
45  // present in .NET assembly. For that reason, messages are not
46  // automatically installed when decoding the Unity class.
47  //
48  // This function can be called by the iCanScript user to add to the
49  // existing Unity library.
50  //
51  public static void InstallUnityMessage(Type classType, string messageName, iCS_StorageClass storageClass,
52  iCS_Parameter[] parameters, iCS_FunctionReturn functionReturn,
53  string iconPath= null, string description= null) {
54  if(iconPath == null) iconPath= kUnityIcon;
55  if(description == null) description= "Event: "+messageName+" on "+classType.Name;
56  iCS_LibraryDatabase.AddMessage(classType, messageName, storageClass, parameters, functionReturn, description, iconPath);
57  }
58  // ----------------------------------------------------------------------
59  // Use this function to register a Unity message on the

61  //
62  // This function can be called by the iCanScript user to add to the
63  // existing Unity library.
64  //
65  public static void InstallMonoBehaviourMessage(string messageName, iCS_Parameter[] parameters,
66  string iconPath= null, string description= null) {
67  var voidReturn= new iCS_FunctionReturn("", typeof(void));
68  InstallUnityMessage(typeof(MonoBehaviour), messageName, iCS_StorageClass.Instance, parameters, voidReturn, iconPath, description);
69  }
