341  (port.IsInputPort && (newNodeParent == portParentNode || portParentNode.IsParentOf(newNodeParent)))) {

504  void ProcessMenu(object obj) {
505  iCS_MenuContext context= obj as iCS_MenuContext;
506  iCS_EditorObject targetObject= context.SelectedObject;
507  Vector2 globalPos= context.GraphPosition;
508  iCS_IStorage iStorage= context.Storage;
509  // Process all other types of requests.
510  switch(context.HiddenCommand) {
511  case SetAsDisplayRootStr: iCS_UserCommands.SetAsDisplayRoot(targetObject); break;
512  case ClearNavigationHistoryStr: iCS_UserCommands.ResetDisplayRoot(iStorage); break;
513  case PackageStr: iCS_UserCommands.CreatePackage(targetObject, globalPos, null); break;
514  case StateChartStr: iCS_UserCommands.CreateStateChart(targetObject, globalPos, null); break;
515  case StateStr: iCS_UserCommands.CreateState(targetObject, globalPos, null); break;
516  case SetAsEntryStr: iCS_UserCommands.SetAsStateEntry(targetObject); break;
517  case OnEntryStr: iCS_UserCommands.CreateOnEntryPackage(targetObject, globalPos); break;
518  case OnUpdateStr: iCS_UserCommands.CreateOnUpdatePackage(targetObject, globalPos); break;
519  case OnExitStr: iCS_UserCommands.CreateOnExitPackage(targetObject, globalPos); break;
520  case ObjectInstanceStr: CreateObjectInstance(context); break;
521  case ShowHierarchyStr: iCS_UserCommands.ShowInHierarchy(targetObject); break;
522  case DeleteStr: iCS_UserCommands.DeleteObject(targetObject); break;
523  case EnablePortStr: iCS_UserCommands.CreateEnablePort(targetObject); break;
524  case TriggerPortStr: iCS_UserCommands.CreateTriggerPort(targetObject); break;
525  case OutputInstancePortStr: iCS_UserCommands.CreateOutInstancePort(targetObject); break;
526  case WrapInPackageStr: iCS_UserCommands.WrapInPackage(targetObject); break;
527  case UpdateMessageHandlerPortsStr: iCS_UserCommands.UpdateMessageHandlerPorts(targetObject); break;
528  case RemoveUnusedPortsStr: iCS_UserCommands.RemoveUnusedPorts(targetObject); break;
529  case MultiSelectionWrapInPackageStr: iCS_UserCommands.WrapMultiSelectionInPackage(iStorage); break;
530  case MultiSelectionDeleteStr: iCS_UserCommands.DeleteMultiSelectedObjects(iStorage); break;
531  case UnwrapPackageStr: iCS_UserCommands.DeleteKeepChildren(targetObject); break;
532  case UserMessageHandlerStr: iCS_UserCommands.CreateUserMessageHandler(targetObject, globalPos); break;
533  default: {
534  iCS_MethodBaseInfo desc= context.Descriptor;
535  if(desc == null) {
536  Debug.LogWarning(iCS_Config.ProductName+": Can find reflection descriptor to create node !!!");
537  break;
538  }
539  if(targetObject == null) break;
540  if(desc.IsMessage) {
541  iCS_UserCommands.CreateMessageHandler(targetObject, globalPos, desc as iCS_MessageInfo);
542  } else {
543  if(targetObject.IsPort) {
544  CreateAttachedMethod(targetObject, iStorage, globalPos, desc);
545  }
546  else {
547  iCS_UserCommands.CreateFunction(targetObject, globalPos, desc);

575  static iCS_EditorObject CreateAttachedMethod(iCS_EditorObject port, iCS_IStorage iStorage, Vector2 globalPos, iCS_MethodBaseInfo desc) {
576  iCS_EditorObject newNodeParent= iStorage.GetNodeAt(globalPos);
577  if(newNodeParent == null) return null;
578  if(!newNodeParent.IsKindOfPackage || newNodeParent.IsBehaviour) return null;
579  // FIXME: Animation & force layout sometime conflict.
580  // Open a transaction for multi-operations
581  iCS_UserCommands.OpenTransaction(iStorage);
582  iCS_EditorObject method= null;
583  iStorage.AnimateGraph(null,
584  _=> {
585  if(newNodeParent.IsInstanceNode) {
586  method= iCS_UserCommands.CreateInstanceWizardElement(newNodeParent, desc);
587  }
588  else {
589  bool createMethod= true;
590  if(desc.IsInstanceFunction || desc.IsInstanceField) {
591  if(desc.ClassType != port.RuntimeType) {
592  int sel= EditorUtility.DisplayDialogComplex("Missing the Instance Node",
593  "The function you selected requires an instance.\nPlease select one of the following:\n1) create the Instance Builder and Instance Node;\n2) create the Instance Node (binding of the instance will be needed);\n3) create the Function. (binding of the instance will be needed).",
594  "Function",
595  "Build Instance",
596  "Instance Node");
597  switch(sel) {
598  case 0:
599  break;
600  case 1: {
601  method= iCS_UserCommands.CreateInstanceBuilderAndObjectAndElement(newNodeParent, globalPos, desc.ClassType, desc);
602  createMethod= false;
603  break;
604  }
605  case 2: {
606  method= iCS_UserCommands.CreateInstanceObjectAndElement(newNodeParent, globalPos, desc.ClassType, desc);
607  createMethod= false;
608  break;
609  }}
610  }
611  }
612  if(createMethod) {
613  method= iCS_UserCommands.CreateFunction(newNodeParent, globalPos, desc);
614  }
615  }
616 
617  // Inverse effective data flow if new node is inside port parent.
618  bool isConsumerPort= port.IsInputPort;
619  var portParent= port.ParentNode;
620  if(portParent.IsParentOf(method)) isConsumerPort= !isConsumerPort;
621  iCS_EditorObject attachedPort= null;
622  iCS_EditorObject providerPort= null;
623  iCS_EditorObject consumerPort= null;
624  if(isConsumerPort) {
625  iCS_EditorObject[] outputPorts= Prelude.filter(x=> iCS_Types.IsA(port.RuntimeType, x.RuntimeType), iStorage.GetChildOutputDataPorts(method));
626  // Connect if only one possibility.
627  if(outputPorts.Length == 1) {
628  attachedPort= outputPorts[0];
629  consumerPort= port;
630  providerPort= attachedPort;
631  }
632  else {
633  var bestPort= GetClosestMatch(port, outputPorts);
634  if(bestPort != null) {
635  attachedPort= bestPort;
636  consumerPort= port;
637  providerPort= bestPort;
638  }
639  }
640  } else {
641  iCS_EditorObject[] inputPorts= Prelude.filter(x=> iCS_Types.IsA(x.RuntimeType, port.RuntimeType), iStorage.GetChildInputDataPorts(method));
642  // Connect if only one posiibility
643  if(inputPorts.Length == 1) {
644  attachedPort= inputPorts[0];
645  consumerPort= attachedPort;
646  providerPort= port;
647  }
648  // Multiple choices exist so try the one with the closest name.
649  else {
650  var bestPort= GetClosestMatch(port, inputPorts);
651  if(bestPort != null) {
652  attachedPort= bestPort;
653  consumerPort= bestPort;
654  providerPort= port;
655  }
656  }
657  }
658  // Position attached port and layout binding.
659  if(attachedPort != null && consumerPort != null && providerPort != null) {
660  iStorage.AutoLayoutPort(attachedPort, port.GlobalPosition, attachedPort.ParentNode.GlobalPosition);
661  iStorage.SetAndAutoLayoutNewDataConnection(consumerPort, providerPort);
662  }
663  iStorage.ForcedRelayoutOfTree();
664  }
665  );
666  iCS_UserCommands.CloseTransaction(iStorage, "Create => "+desc.DisplayName);
667  return method;
668  }

674  static iCS_EditorObject GetClosestMatch(iCS_EditorObject refPort, iCS_EditorObject[] otherPorts) {
675  if(otherPorts.Length == 0) return null;
676  float bestScore= -1f;
677  iCS_EditorObject bestPort= null;
678  string refName= refPort.Name;
679  if(string.IsNullOrEmpty(refPort.RawName)) refName= refPort.ParentNode.Name;
680  foreach(var p in otherPorts) {
681  string pName= p.Name;
682  if(string.IsNullOrEmpty(p.RawName)) pName= p.ParentNode.Name;
683  var score= iCS_StringUtil.FuzzyCompare(refName, pName);
684  if(score > bestScore) {
685  bestScore= score;
686  bestPort= p;
687  }
688  }
689  return bestPort;
690  }
