246  // Resize destination engine object array.
247  var fromEngineObjects= from.EngineObjects;
248  var toEngineObjects= to.EngineObjects;
249  int fromLen= fromEngineObjects.Count;
250  int toLen= toEngineObjects.Count;
251  if(toLen > fromLen) {
252  toEngineObjects.RemoveRange(fromLen, toLen-fromLen);
253  }
254  toEngineObjects.Capacity= fromLen;
255  // Copy engine objects.
256  for(int i= 0; i < fromLen; ++i) {
257  var fromObj= fromEngineObjects[i];
258  if(fromObj == null) fromObj= iCS_EngineObject.CreateInvalidInstance();
259  if(toEngineObjects.Count <= i) {
260  toEngineObjects.Add(fromObj.Clone());
261  }
262  else if(toEngineObjects[i] == null) {
263  toEngineObjects[i]= fromObj.Clone();
264  }
265  else {
266  toEngineObjects[i]= fromObj.CopyTo(toEngineObjects[i]);
267  }
268  }
269  // Resize Unity object reference array
270  var fromUnityObjects= from.UnityObjects;
271  var toUnityObjects= to.UnityObjects;
272  fromLen= fromUnityObjects.Count;
273  toLen= toUnityObjects.Count;
274  if(toLen > fromLen) {
275  toUnityObjects.RemoveRange(fromLen, toLen-fromLen);
276  }
277  toUnityObjects.Capacity= fromLen;
278  // Copy Unity Object references.
279  for(int i= 0; i < fromLen; ++i) {
280  var fromObj= fromUnityObjects[i];
281  if(toUnityObjects.Count <= i) {
282  toUnityObjects.Add(fromObj);
