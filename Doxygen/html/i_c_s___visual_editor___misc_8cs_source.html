296 #if DEBUG
297  Debug.Log("iCanScript: RebuildDataConnection: output= "+outputPort.Name+" input= "+inputPort.Name);

299  // Have we completed rebuilding ... if so return.
300  if(inputPort == outputPort) return;
301  var inputNode= inputPort.ParentNode;
302  var outputNode= outputPort.ParentNode;
303  if(inputNode == outputNode) return;
304  // outputPort is inside the node with the inputPort.
305  var commonParentNode= outputPort.GetCommonParent(inputPort);
306  if(inputNode == commonParentNode) {
307  // Rebuild moving down from the common parent towards the output port.
308  var newInputNode= outputPort.ParentNode;
309  while(newInputNode != inputNode && newInputNode.ParentNode != inputNode) {
310  newInputNode= newInputNode.ParentNode;
311  }
312  var existingPort= IStorage.FindPortWithSourceEndPoint(newInputNode, outputPort);
313  if(existingPort != null) {
314  var prevSource= inputPort.ProviderPort;
315  if(prevSource != existingPort) {
316  inputPort.ProviderPort= existingPort;
317  if(prevSource.IsDynamicDataPort && !inputPort.IsPartOfConnection(prevSource)) {
318  IStorage.CleanupHangingConnection(prevSource);
319  }
320  }
321  RebuildDataConnection(outputPort, existingPort);
322  } else {
323  iCS_EditorObject newPort= IStorage.CreatePort(inputPort.Name, newInputNode.InstanceId, inputPort.RuntimeType, iCS_ObjectTypeEnum.OutDynamicDataPort);
324  IStorage.SetBestPositionForAutocreatedPort(newPort, outputPort.GlobalPosition, inputPort.GlobalPosition);
325  newPort.ProviderPort= inputPort.ProviderPort;

327  RebuildDataConnection(outputPort, newPort);
328  }
329  return;
330  }
331  var inputNodeParent= inputNode.ParentNode;
332  if(inputNodeParent == commonParentNode) {
333  // Rebuild traversing from moving upwards to downwords.
