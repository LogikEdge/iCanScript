115  // returned is the smallest distance to remove the overlap.
116  Vector2 GetSeperationVector(iCS_EditorObject theOther, Rect myRect, Rect otherRect) {
117  myRect= AddMargins(myRect);
118  // No collision if X & Y distance of the enclosing rect is either
119  // larger or higher then the total width/height.
120  var intersection= Math3D.Intersection(myRect, otherRect);
121  if(Math3D.IsSmallerOrEqual(intersection.width,0f) || Math3D.IsSmallerOrEqual(intersection.height, 0f)) {
122  return Vector2.zero;
123  }
124  // A collision is detected. The seperation vector is the
125  // smallest distance to remove the overlap. The separtion
126  // must also respect the anchor position relationship
127  // between the two overalpping nodes.

130  var pivotSepDir= theOtherPivot-myPivot;
131  if(Mathf.Abs(pivotSepDir.x) > Mathf.Abs(pivotSepDir.y)) {
132  return new Vector2(intersection.width*Mathf.Sign(pivotSepDir.x), 0f);
