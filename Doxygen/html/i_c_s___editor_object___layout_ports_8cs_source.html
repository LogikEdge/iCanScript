75  static float[] ResolvePortCollisions(float[] sortedPosition, float availableLength) {
76  // Nothing collision to resolve if we don't have at least two ports.
77  int nbPorts= sortedPosition.Length;
78  if(nbPorts < 2) return sortedPosition;

80  // In such a situation, we simply distribute the ports has best we can on the
81  // available size.
82  float minSeparation= iCS_EditorConfig.MinimumPortSeparation;
83  if(minSeparation*(nbPorts-1) >= availableLength) {
84  var step= availableLength/(nbPorts-1);
85  for(int i= 0; i < nbPorts; ++i) {
86  sortedPosition[i]= i*step;
87  }
88  return sortedPosition;
89  }
90  // We have enough space to resolve the port layout and avoid overlap.
91  bool mustRelayout= true;
92  while(mustRelayout) {
93  mustRelayout= false;
94  for (int i= 0; i < nbPorts-1; ++i) {
95  var p1= sortedPosition[i];
96  int j= i+1;
97  var p2= sortedPosition[j];
98  var diff= p2-p1;
99  if(Math3D.IsGreater(diff, minSeparation)) continue;
100  mustRelayout= Math3D.IsSmaller(diff, minSeparation);
101  for (; j < nbPorts-1; ++j) {
102  p1= p2;
103  p2= sortedPosition[j+1];
104  diff= p2-p1;
105  if(Math3D.IsGreater(diff, minSeparation)) break;
106  if(Math3D.IsSmaller(diff, minSeparation)) {
107  mustRelayout= true;
108  }
109  }
110  if(mustRelayout) {
111  var iPos= sortedPosition[i];
112  var jPos= sortedPosition[j];

115  var halfDelta= 0.5f*(minSep-diff);
116  iPos= iPos-halfDelta;

118  if(iPos < 0) {
119  jPos+= -iPos;

121  }
122  if(jPos > availableLength) {
123  iPos-= jPos-availableLength;
124  }
125  for(int k= 0; k <= (j-i); ++k) {
126  sortedPosition[i+k]= iPos+k*minSeparation;
127  }

129  }
130  }
