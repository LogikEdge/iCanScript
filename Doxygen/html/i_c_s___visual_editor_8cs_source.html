165  public void Update() {
166  // Don't run update faster then requested.
167  float currentTime= Time.realtimeSinceStartup;
168  int newUpdateCounter= (int)(currentTime*kUpdateRate);
169  if(newUpdateCounter == myUpdateCounter) return;
170  myUpdateCounter= newUpdateCounter;
171 
172  // Abort if our environment is not initialized.
173  if(!IsInitialized()) return;
174 
175  // Determine repaint rate.
176  if(IStorage != null) {
177  // Update DisplayRoot
178  if(DisplayRoot == null && IStorage.IsValid(0)) {
179  DisplayRoot= IStorage[0];
180  }
181 
182  // Repaint visual editor if it has changed
183  if(IStorage.IsAnimationPlaying || myAnimatedScrollPosition.IsActive || myAnimatedScale.IsActive || mySubEditor != null) {
184  Repaint();
185  myNeedRepaint= true;
186  }
187  // Repaint on request.
188  else if(myNeedRepaint) {
189  Repaint();
190  myNeedRepaint= false;
191  }
192  // Repaint if game is running.
193  else if(Application.isPlaying && Prefs.ShowRuntimePortValue) {
194  float period= Prefs.PortValueRefreshPeriod;
195  if(period < 0.1f) period= 0.1f;
196  float refreshFactor= 1f/period;
197  int newRefreshCounter= (int)(currentTime*refreshFactor);
198  if(newRefreshCounter != myRefreshCounter) {
199  myRefreshCounter= newRefreshCounter;
200  Repaint();
201  }
202  }
203 #if FORCE_REPAINT
204  else {
205  Repaint();
206  }
207 #endif
208  }
209 
210  // Cleanup memory pool.
211  iCS_AutoReleasePool.Update();
212  }
