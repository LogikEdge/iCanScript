142  index= GetNextAvailableEnablePortIndex(parent);

144  else {

147  }
148  iCS_EditorObject port= iCS_EditorObject.CreateInstance(id, name, valueType, parentId, portType, this);
149  port.PortIndex= index;

157  }
158  // Set initial port edge.

160  port.Edge= iCS_EdgeEnum.Top;
161  } else if(port.IsTriggerPort) {
162  port.Edge= iCS_EdgeEnum.Bottom;

164  port.Edge= iCS_EdgeEnum.Left;

189  public iCS_EditorObject[] RecalculatePortIndexes(iCS_EditorObject node) {
190  List<iCS_EditorObject> ports= new List<iCS_EditorObject>();
191  // Get all child data ports.
192  ForEachChildDataPort(node, child=> ports.Add(child));
193  // Sort child ports according to index.
194  iCS_EditorObject[] result= SortPortsOnIndex(ports.ToArray());
195  // Find first dynamic or proposed port
196  int firstDynamicIdx= 0;
197  for(int i= 0; i < result.Length; ++i) {
198  if(result[i].IsParameterPort && result[i].IsFixDataPort) {
199  firstDynamicIdx= result[i].PortIndex+1;
200  }
201  }
202  // Re-index dynamic and proposed ports
203  for(int i= 0; i < result.Length; ++i) {
204  if(result[i].IsDynamicDataPort || result[i].IsProposedDataPort) {
205  if(result[i].PortIndex <= (int)iCS_PortIndex.ParametersEnd) {
206  result[i].PortIndex= firstDynamicIdx++;
207  }
208  }
209  }
210  return result;
211  }

233  public static iCS_EditorObject[] SortPortsOnIndex(iCS_EditorObject[] lst) {
234  // Find largest dynamic or proposed port.
235  int lastIndex= -1;
236  foreach(var p in lst) {
237  if(p.IsDynamicDataPort || p.IsProposedDataPort) {
238  if(p.PortIndex > lastIndex) {
239  lastIndex= p.PortIndex;
240  }
241  }
242  }
243  // Assign all unassigned port indexes (we assume that it is a dynmic port).
244  if(lastIndex != -1) {
245  foreach(var p in lst) {
246  if(p.PortIndex < 0) {
247  p.PortIndex= ++lastIndex;
248  }
249  }
250  }
251  Array.Sort(lst, (x,y)=> x.PortIndex - y.PortIndex);
