53  public static Texture2D GetIconFor(iCS_EditorObject obj) {
54  // No default icon for null object.
55  if(obj == null) {
56  return null;
57  }
58  // Product icons cannot be changed.
59  Texture2D icon= null;
60  if(obj.IsOnStateEntryPackage) {
61  icon= iCS_TextureCache.GetIcon(kOnStateEntryIcon);
62  if(icon != null) return icon;
63  } else if(obj.IsOnStateUpdatePackage) {
64  icon= iCS_TextureCache.GetIcon(kOnStateUpdateIcon);
65  if(icon != null) return icon;
66  } else if(obj.IsOnStateExitPackage) {
67  icon= iCS_TextureCache.GetIcon(kOnStateExitIcon);
68  if(icon != null) return icon;
69  }
70  // Use user defined icon if it exists ... otherwise use default for the
71  // given type of node.
72  if(obj.IconGUID != null) {
73  icon= iCS_TextureCache.GetIconFromGUID(obj.IconGUID);
74  if(icon != null) return icon;
75  }
76  return GetDefaultNodeIconFor(obj);
77  }
