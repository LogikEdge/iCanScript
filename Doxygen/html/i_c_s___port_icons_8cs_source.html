243  public static void BuildMuxPortTemplate(float width, float height, ref Texture2D texture, bool isInPort, float rotation= 0f) {
244  // Compute texture size
245  Vector2[] muxPolygon= null;
246  int textureWidth= (int)(width+3f);
247  int textureHeight= (int)(height+3f);
248  if(Math3D.IsEqual(rotation, 0f)) {
249  if(isInPort) {
250  muxPolygon= Math3D.FlipPolygonVertically(myMuxPortPolygon, 0f);
251  } else {
252  muxPolygon= myMuxPortPolygon;
253  }
254  } else if(Math3D.IsEqual(rotation, 90f)) {
255  if(isInPort) {
256  muxPolygon= Math3D.Rotate90DegreesPolygon(myMuxPortPolygon);
257  muxPolygon= Math3D.FlipPolygonHorizontally(muxPolygon, 0f);
258  } else {
259  muxPolygon= Math3D.Rotate90DegreesPolygon(myMuxPortPolygon);
260  }
261  textureWidth= (int)(height+3f);
262  textureHeight= (int)(width+3f);
263  } else if(Math3D.IsEqual(rotation, 180f)) {
264  if(isInPort) {
265  muxPolygon= myMuxPortPolygon;
266  } else {
267  muxPolygon= Math3D.FlipPolygonVertically(myMuxPortPolygon, 0f);
268  }
269  } else {
270  if(isInPort) {
271  muxPolygon= Math3D.Rotate90DegreesPolygon(myMuxPortPolygon);
272  } else {
273  muxPolygon= Math3D.Rotate90DegreesPolygon(myMuxPortPolygon);
274  muxPolygon= Math3D.FlipPolygonHorizontally(muxPolygon, 0f);
275  }
276  textureWidth= (int)(height+3f);
277  textureHeight= (int)(width+3f);
278  }
279  // Allocate texture
280  if(texture != null) Texture2D.DestroyImmediate(texture);
281  texture= new Texture2D(textureWidth, textureHeight, TextureFormat.ARGB32, false);
282  iCS_TextureUtil.Clear(ref texture);
283  // Draw black background polygon.
284  var textureCenter= new Vector2(0.5f*textureWidth,0.5f*textureHeight);
285  var outterPolygon= Math3D.ScaleAndTranslatePolygon(muxPolygon, new Vector2(height,height), textureCenter);
286  iCS_TextureUtil.DrawFilledPolygon(ref texture, outterPolygon, Color.black);
287  // Draw inner color polygon.
288  var innerPolygon= Math3D.ScaleAndTranslatePolygon(muxPolygon, new Vector2(0.6f*height, 0.6f*height), textureCenter);
289  if(isInPort) {
290  iCS_TextureUtil.DrawPolygonOutline(ref texture, innerPolygon, Color.red);
291  } else {
292  iCS_TextureUtil.DrawFilledPolygon(ref texture, innerPolygon, Color.red);
293  }
294  // Finalize texture.
295  texture.hideFlags= HideFlags.DontSave;
296  texture.Apply();
297  }

512  public static Texture2D BuildPortIcon(Color typeColor, Texture2D iconTemplate) {
513  int width= iconTemplate.width;
514  int height= iconTemplate.height;
515  Texture2D icon= new Texture2D(width, height);
516  for(int x= 0; x < width; ++x) {
517  for(int y= 0; y < height; ++y) {
518  Color pixel= iconTemplate.GetPixel(x,y);
519  if(pixel.r == 0) {
520  icon.SetPixel(x,y, pixel);
521  } else {
522  // Anti-Aliasing fill.
523  Color c;
524  c.r= pixel.r*typeColor.r;
525  c.g= pixel.r*typeColor.g;
526  c.b= pixel.r*typeColor.b;
527  c.a= pixel.a;
528  icon.SetPixel(x,y, c);
529  }
530  }
531  }
532  icon.Apply();
533  icon.hideFlags= HideFlags.DontSave;
534  return icon;
