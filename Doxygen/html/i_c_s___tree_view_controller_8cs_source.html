101  void BuildTree() {
102  // Build filter list of object...
103  var filterFlags= Prelude.map(o=> FilterIn(o), myIStorage.EditorObjects);
104  // ... make certain the parents are also filtered in !!!
105  myTreeSize= 0;
106  for(int i= 0; i < filterFlags.Count; ++i) {
107  if(filterFlags[i]) {
108  ++myTreeSize;
109  Prelude.until(
110  myIStorage.IsValid,
111  id=> {
112  if(myIStorage.IsIdValid(id)) {
113  filterFlags[id]= true;
114  return myIStorage.EditorObjects[id].ParentId;
115  }
116  filterFlags[id]= false;
117  return -1;
118  },
119  myIStorage.IsIdValid(i) ? myIStorage.EditorObjects[i].ParentId : -1
120  );
121  }
122  }
123  // Build tree and sort it elements.
124  if(filterFlags.Count == 0 || !filterFlags[0]) {
125  myTree= null;
126  return;
127  }
128  var rootNode= myIStorage.EditorObjects[0];
129  myTree= BuildTreeNode(rootNode, filterFlags);
130  }
131  Prelude.Tree<iCS_EditorObject> BuildTreeNode(iCS_EditorObject nodeRoot, List<bool> filterFlags) {
132  Prelude.Tree<iCS_EditorObject> tree= new Prelude.Tree<iCS_EditorObject>(nodeRoot);
133  myIStorage.ForEachChild(nodeRoot,
134  c=> {
135  Prelude.Tree<iCS_EditorObject> newNode= BuildTreeNode(c, filterFlags);
136  if(filterFlags[c.InstanceId]) tree.AddChild(newNode);
137  }
138  );
139  tree.Sort(SortComparaison);
140  return tree;
141  }
142  int SortComparaison(iCS_EditorObject x, iCS_EditorObject y) {
143  if(x.IsInputPort && !y.IsInputPort) return -1;
144  if(y.IsInputPort && !x.IsInputPort) return 1;
145  if(x.IsOutputPort && !y.IsOutputPort) return -1;
146  if(y.IsOutputPort && !x.IsOutputPort) return 1;
147  if(x.IsInstanceNode && !y.IsInstanceNode) return -1;
148  if(y.IsInstanceNode && !x.IsInstanceNode) return 1;
149  return String.Compare(x.Name, y.Name);
150  }
