20  if(IsSorted) return;

22  int cmpCnt= 0;
23  int len= Functions.Count;
24  int step= (len >> 1) + (len & 1);

34  Functions[j]= tmp;
35  int k= i-step;
36  while(k >= 0) {
37  ++cmpCnt;
38  if(CompareFunctionNames(Functions[k], Functions[k+step]) < 0) break;
39  ++reorderCnt;
40  tmp= Functions[k];
41  Functions[k]= Functions[k+step];
42  Functions[k+step]= tmp;
43  k-= step;
44  }

46  ++i;
47  ++j;
48  }
49  step >>= 1;
50  }
51  IsSorted= true;
52  }
