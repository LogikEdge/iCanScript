346  public static iCS_EditorObject CreateTransition(iCS_EditorObject fromStatePort, iCS_EditorObject toState, Vector2 toStatePortPos) {
347 #if DEBUG
348  Debug.Log("iCanScript: Create Transition Package");
349 #endif
350  if(fromStatePort == null || toState == null) return null;
351  if(!IsCreationAllowed()) return null;
352  var iStorage= toState.IStorage;
353  OpenTransaction(iStorage);
354  iCS_EditorObject transitionPackage= null;
355  try {
356  // Create toStatePort
357  iCS_EditorObject toStatePort= iStorage.CreatePort("", toState.InstanceId, typeof(void), iCS_ObjectTypeEnum.InStatePort);
358  // Update port positions
359  toStatePort.SetInitialPosition(toStatePortPos);
360  toStatePort.UpdatePortEdge();
361  fromStatePort.UpdatePortEdge();
362  // Temporally connect state ports together.
363  iStorage.SetSource(toStatePort, fromStatePort);
364  // Create transition package
365  transitionPackage= iStorage.CreateTransition(fromStatePort, toStatePort);
366  // Try to position the transition in the middle
367  var fromStatePortPos= fromStatePort.GlobalPosition;
368  var globalPos= 0.5f*(fromStatePortPos+toStatePortPos);
369  transitionPackage.SetInitialPosition(globalPos);
370  transitionPackage.Iconize();
371  // Attempt to proper edge for transition ports.
372  var outTransitionPort= toStatePort.ProviderPort;
373  var inTransitionPort= iStorage.GetInTransitionPort(transitionPackage);
374  var diff= toStatePortPos-fromStatePortPos;
375  if(Mathf.Abs(diff.x) > Mathf.Abs(diff.y)) {
376  if(Vector2.Dot(diff, Vector2.right) > 0) {
377  inTransitionPort.Edge= iCS_EdgeEnum.Left;
378  toStatePort.Edge= iCS_EdgeEnum.Left;
379  outTransitionPort.Edge= iCS_EdgeEnum.Right;
380  fromStatePort.Edge= iCS_EdgeEnum.Right;
381  }
382  else {
383  inTransitionPort.Edge= iCS_EdgeEnum.Right;
384  toStatePort.Edge= iCS_EdgeEnum.Right;
385  outTransitionPort.Edge= iCS_EdgeEnum.Left;
386  fromStatePort.Edge= iCS_EdgeEnum.Left;
387  }
388  }
389  else {
390  if(Vector2.Dot(diff, Vector2.up) > 0) {
391  inTransitionPort.Edge= iCS_EdgeEnum.Top;
392  toStatePort.Edge= iCS_EdgeEnum.Top;
393  outTransitionPort.Edge= iCS_EdgeEnum.Bottom;
394  fromStatePort.Edge= iCS_EdgeEnum.Bottom;
395  }
396  else {
397  inTransitionPort.Edge= iCS_EdgeEnum.Bottom;
398  toStatePort.Edge= iCS_EdgeEnum.Bottom;
399  outTransitionPort.Edge= iCS_EdgeEnum.Top;
400  fromStatePort.Edge= iCS_EdgeEnum.Top;
401  }
402  }
403  inTransitionPort.PortPositionRatio= 0.5f;
404  outTransitionPort.PortPositionRatio= 0.5f;
405  // Layout the graph
406  iStorage.ForcedRelayoutOfTree();
407  iStorage.ReduceCollisionOffset();
408  }
409  catch(System.Exception) {
410  CancelTransaction(iStorage);
411  return null;
412  }
413  if(transitionPackage == null) {
414  CancelTransaction(iStorage);
415  return null;
416  }
417  CloseTransaction(iStorage, "Create Transition");
418  return transitionPackage;
419  }
