408  // This attempt to properly locate an autocreated data port.
409  public void SetBestPositionForAutocreatedPort(iCS_EditorObject port, Vector2 inPortPosition, Vector2 outPortPosition) {
410  // Determine the parent edge position to use.
411  var parent= port.Parent;
412  var parentGlobalRect= parent.GlobalRect;
413  float x= port.IsInputPort ? parentGlobalRect.xMin : parentGlobalRect.xMax;
414  // Assure that the in position X value is smaller then the out position.
415  if(inPortPosition.x > outPortPosition.x) {
416  var tmp= inPortPosition; inPortPosition= outPortPosition; outPortPosition= tmp;
417  }
418  // Manage situation where new port is between the in & out ports.
419  var parentGlobalPosition= Math3D.Middle(parentGlobalRect);
420  var top = parentGlobalPosition.y+parent.VerticalPortsTop;
421  var bottom= parentGlobalPosition.y+parent.VerticalPortsBottom;
422  float y;
423  if(Math3D.IsSmaller(inPortPosition.x, x) && Math3D.IsGreater(outPortPosition.x, x)) {
424  float ratio= (x-inPortPosition.x)/(outPortPosition.x-inPortPosition.x);
425  y= Math3D.Lerp(inPortPosition.y, outPortPosition.y, ratio);
426  if(y < top) {
427  y= top;
428  }
429  if(y > bottom) {
430  y= bottom;
431  }
432  port.LocalAnchorFromGlobalPosition= new Vector2(x,y);
433  return;
434  }
435  if(Math3D.IsEqual(inPortPosition.y, outPortPosition.y)) {
436  port.LocalAnchorFromGlobalPosition= new Vector2(x, 0.5f*(top+bottom));
437  return;
438  }
439  // Assure that the in position Y value is smaller then the out position.
440  if(inPortPosition.y > outPortPosition.y) {
441  var tmp= inPortPosition; inPortPosition= outPortPosition; outPortPosition= tmp;
442  }
443  // Compute some type of ratio if Y position traverse the top port position
444  if(Math3D.IsSmaller(inPortPosition.y, top) && Math3D.IsGreater(outPortPosition.y, top)) {
445  float y1= outPortPosition.y-top;
446  float y2= top-inPortPosition.y;
447  y= top+(y1*y1/(y1+y2));
448  } else {
449  float y2= outPortPosition.y-bottom;
450  float y1= bottom-inPortPosition.y;
451  y= bottom-(y1*y1/(y1+y2));
452  }
453  port.LocalAnchorFromGlobalPosition= new Vector2(x,y);
454  return;
455  }
