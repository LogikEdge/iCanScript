108  Rect NodeRectFromChildrenRectWithMargins(Rect childRect) {
109  // Get padding for all sides.
110  float topPadding= NodeTopPadding;
111  float bottomPadding= NodeBottomPadding;
112  float leftPadding= NodeLeftPadding;
113  float rightPadding= NodeRightPadding;
114  // Determine size of node.
115  float width = childRect.width+leftPadding+rightPadding;
116  float height= childRect.height+topPadding+bottomPadding;
117  // Assure minimum size for title and ports.
118  var titleHeight= NodeTitleHeight;
119  var titleWidth= NodeTitleWidth;
120  var neededPortHeight= MinimumHeightForPorts;
121  var neededPortWidth = MinimumWidthForPorts;
122  var minHeight= titleHeight+neededPortHeight;
123  var minWidth= Mathf.Max(titleWidth, neededPortWidth);
124  // Readjust parent size & position.
125  float xOffset= 0f;
126  float yOffset= 0f;
127  if(width < minWidth) {
128  xOffset= 0.5f*(minWidth-width);
129  width= minWidth;
130  }
131  if(height < minHeight) {
132  yOffset= 0.5f*(minHeight-height);
133  height= minHeight;
134  }
135  // Determine rect to wrap children.
136  float x, y;
137  if(Math3D.IsZero(childRect.width) || Math3D.IsZero(childRect.height)) {
138  var pos= GlobalPosition;
139  x= pos.x-0.5f*width;
140  y= pos.y-0.5f*height;
141  } else {
142  x= childRect.x-leftPadding-xOffset;
143  y= childRect.y-topPadding-yOffset;
144  }
145  var r= new Rect(x, y, width,height);
146  return r;
147  }
