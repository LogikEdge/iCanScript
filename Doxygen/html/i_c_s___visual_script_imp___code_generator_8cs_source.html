17  bool storageCorruption= false;

19  do {
20  sanityNeeded= false;

22  if(EngineObjects[i].InstanceId == -1) continue;
23  if(EngineObjects[i].InstanceId != i) {
24  sanityNeeded= true;
25  Debug.LogWarning("iCanScript: Sanity Check: Object: "+i+" has an invalid instance id of: "+EngineObjects[i].InstanceId);
26  EngineObjects[i].Reset();
27  continue;
28  }
29  int parentId= EngineObjects[i].ParentId;
30  if(i == 0) {
31  if(parentId != -1) {
32  sanityNeeded= true;
33  EngineObjects[0].ParentId= -1;
34  Debug.LogWarning("iCanScript: Sanity: Root object has a parent of: "+parentId);
35  continue;
36  }
37  } else {
38  // The parent id must be valid.
39  if(!IsInBounds(parentId)) {
40  sanityNeeded= true;
41  EngineObjects[i].Reset();
42  Debug.LogWarning("iCanScript Sanity: Object: "+i+" has an invalid parent: "+parentId);
43  continue;
44  }
45  // A port cannot be a parent.
46  if(EngineObjects[parentId].IsPort && !EngineObjects[parentId].IsParentMuxPort) {
47  sanityNeeded= true;
48  EngineObjects[i].Reset();
49  Debug.LogWarning("iCanScript Sanity: Port: "+parentId+" is the parent of: "+i);
50  continue;
51  }
52  }
53  }
54  if(sanityNeeded) {
55  storageCorruption= true;
56  Debug.LogWarning("iCanScript: Sanity detected an error.");
57  }
58  } while(sanityNeeded);
